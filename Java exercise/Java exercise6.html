Logical Operators

AND (&&)

  1. Write a Java program to check if a number is between 10 and 20 (inclusive) using the AND operator. 
                            Sample Output
                        
                            Enter a number: 15
                            The number is between 10 and 20.
                        
                    
                                
                                import java.util.Scanner;

                                public class AndOperator1 {
                                    public static void main(String[] args) {
                                        Scanner scanner = new Scanner(System.in);
                                        System.out.print("Enter a number: ");
                                        int num = scanner.nextInt();
                                        if (num >= 10 && num <= 20) {
                                            System.out.println("The number is between 10 and 20.");
                                        } else {
                                            System.out.println("The number is not between 10 and 20.");
                                        }
                                    }
                                }
  2. Write a Java program to check if a character is a vowel (a, e, i, o, u) using the AND operator. 
                            Sample Output
                        
                            Enter a character: e
                            The character is a vowel.
                        
                    
                                
                                import java.util.Scanner;

                                public class AndOperator2 {
                                    public static void main(String[] args) {
                                        Scanner scanner = new Scanner(System.in);
                                        System.out.print("Enter a character: ");
                                        char ch = scanner.next().charAt(0);
                                        if (ch == 'a' && ch == 'e' && ch == 'i' && ch == 'o' && ch == 'u') {
                                            System.out.println("The character is a vowel.");
                                        } else {
                                            System.out.println("The character is not a vowel.");
                                        }
                                    }
                                }
  3. Write a Java program to check if a person is eligible to vote (age >= 18 and is a citizen) using the AND operator. 
                            Sample Output
                        
                            Enter age: 20
                            Are you a citizen (true/false): true
                            You are eligible to vote.
                        
                    
                                
                                import java.util.Scanner;

                                public class AndOperator3 {
                                    public static void main(String[] args) {
                                        Scanner scanner = new Scanner(System.in);
                                        System.out.print("Enter age: ");
                                        int age = scanner.nextInt();
                                        System.out.print("Are you a citizen (true/false): ");
                                        boolean isCitizen = scanner.nextBoolean();

                                        if (age >= 18 && isCitizen) {
                                            System.out.println("You are eligible to vote.");
                                        } else {
                                            System.out.println("You are not eligible to vote.");
                                        }
                                    }
                                }

    4. OR (||)

  4. Write a Java program to check if a number is less than 10 or greater than 20 using the OR operator. 
                            Sample Output
                        
                            Enter a number: 25
                            The number is less than 10 or greater than 20.
                        
                    
                                
                                import java.util.Scanner;

                                public class OrOperator1 {
                                    public static void main(String[] args) {
                                        Scanner scanner = new Scanner(System.in);
                                        System.out.print("Enter a number: ");
                                        int num = scanner.nextInt();
                                        if (num < 10 || num > 20) {
                                            System.out.println("The number is less than 10 or greater than 20.");
                                        } else {
                                            System.out.println("The number is between 10 and 20.");
                                        }
                                    }
                                }

    7. NOT (!)

  7. Write a Java program to check if a number is not equal to 10 using the NOT operator. 
                            Sample Output
                        
                            Enter a number: 15
                            The number is not equal to 10.
                        
                    
                                
                                import java.util.Scanner;

                                public class NotOperator1 {
                                    public static void main(String[] args) {
                                        Scanner scanner = new Scanner(System.in);
                                        System.out.print("Enter a number: ");
                                        int num = scanner.nextInt();
                                        if (num != 10) {
                                            System.out.println("The number is not equal to 10.");
                                        } else {
                                            System.out.println("The number is equal to 10.");
                                        }
                                    }
                                }
  8. Write a Java program to check if a character is not a vowel (a, e, i, o, u) using the NOT operator. 
                            Sample Output
                        
                            Enter a character: k
                            The character is not a vowel.
                        
                    
                                
                                import java.util.Scanner;

                                public class NotOperator2 {
                                    public static void main(String[] args) {
                                        Scanner scanner = new Scanner(System.in);
                                        System.out.print("Enter a character: ");
                                        char ch = scanner.next().charAt(0);
                                        if (ch != 'a' && ch != 'e' && ch != 'i' && ch != 'o' && ch != 'u') {
                                            System.out.println("The character is not a vowel.");
                                        } else {
                                            System.out.println("The character is a vowel.");
                                        }
                                    }
                                }
  9. Write a Java program to check if a person is not eligible to vote (age < 18) using the NOT operator. 
                            Sample Output
                        
                            Enter age: 16
                            You are not eligible to vote.
                        
                    
                                
                                import java.util.Scanner;

                                public class NotOperator3 {
                                    public static void main(String[] args) {
                                        Scanner scanner = new Scanner(System.in);
                                        System.out.print("Enter age: ");
                                        int age = scanner.nextInt();
                                        if (!(age >= 18)) {
                                            System.out.println("You are not eligible to vote.");
                                        } else {
                                            System.out.println("You are eligible to vote.");
                                        }
                                    }
                                }
