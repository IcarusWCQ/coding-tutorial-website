Find the Largest Number in an Array

  1. Write a Java program to find the largest number in an array. 
                            Sample Output
                        
                            Array: [1, 2, 3, 4, 5]
                            The largest number is 5
                        
                    
                                
                                public class LargestNumberInArray {
                                    public static void main(String[] args) {
                                        int[] arr = {1, 2, 3, 4, 5};
                                        int max = arr[0];
                                        for (int i = 1; i < arr.length; i++) {
                                            if (arr[i] > max) {
                                                max = arr[i];
                                            }
                                        }
                                        System.out.println("The largest number is " + max);
                                    }
                                }

    2. Sum of All Elements in an Array

  2. Write a Java program to find the sum of all elements in an array. 
                            Sample Output
                        
                            Array: [1, 2, 3, 4, 5]
                            Sum of all elements: 15
                        
                    
                                
                                public class SumOfArray {
                                    public static void main(String[] args) {
                                        int[] arr = {1, 2, 3, 4, 5};
                                        int sum = 0;
                                        for (int i = 0; i < arr.length; i++) {
                                            sum += arr[i];
                                        }
                                        System.out.println("Sum of all elements: " + sum);
                                    }
                                }

    3. Reverse an Array

  3. Write a Java program to reverse an array. 
                            Sample Output
                        
                            Original array: [1, 2, 3, 4, 5]
                            Reversed array: [5, 4, 3, 2, 1]
                        
                    
                                
                                public class ReverseArray {
                                    public static void main(String[] args) {
                                        int[] arr = {1, 2, 3, 4, 5};
                                        System.out.print("Original array: ");
                                        for (int i = 0; i < arr.length; i++) {
                                            System.out.print(arr[i] + " ");
                                        }
                                        System.out.println();
                                        System.out.print("Reversed array: ");
                                        for (int i = arr.length - 1; i >= 0; i--) {
                                            System.out.print(arr[i] + " ");
                                        }
                                    }
                                }

    4. Find the Second Largest Number in an Array

  4. Write a Java program to find the second largest number in an array. 
                            Sample Output
                        
                            Array: [1, 2, 3, 4, 5]
                            The second largest number is 4
                        
                    
                                
                                public class SecondLargestInArray {
                                    public static void main(String[] args) {
                                        int[] arr = {1, 2, 3, 4, 5};
                                        int largest = arr[0];
                                        int secondLargest = Integer.MIN_VALUE;
                                        for (int i = 1; i < arr.length; i++) {
                                            if (arr[i] > largest) {
                                                secondLargest = largest;
                                                largest = arr[i];
                                            } else if (arr[i] > secondLargest && arr[i] != largest) {
                                                secondLargest = arr[i];
                                            }
                                        }
                                        System.out.println("The second largest number is " + secondLargest);
                                    }
                                }

    5. Remove Duplicates from an Array

  5. Write a Java program to remove duplicate elements from an array. 
                            Sample Output
                        
                            Original array: [1, 2, 2, 3, 4, 4, 5]
                            Array without duplicates: [1, 2, 3, 4, 5]
                        
                    
                                
                                import java.util.Arrays;

                                public class RemoveDuplicatesFromArray {
                                    public static void main(String[] args) {
                                        int[] arr = {1, 2, 2, 3, 4, 4, 5};
                                        Arrays.sort(arr);
                                        int[] temp = new int[arr.length];
                                        int j = 0;
                                        for (int i = 0; i < arr.length - 1; i++) {
                                            if (arr[i] != arr[i + 1]) {
                                                temp[j++] = arr[i];
                                            }
                                        }
                                        temp[j++] = arr[arr.length - 1];

                                        int[] newArr = new int[j];
                                        for (int i = 0; i < j; i++) {
                                            newArr[i] = temp[i];
                                        }

                                        System.out.print("Array without duplicates: ");
                                        for (int i : newArr) {
                                            System.out.print(i + " ");
                                        }
                                    }
                                }
