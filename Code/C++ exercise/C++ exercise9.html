Function Overloading

  1. Write a C++ program to demonstrate function overloading by creating multiple functions with the same name but different parameter lists. 
                            Sample Output
                        
                            Sum of integers: 10
                            Sum of doubles: 10.5
                        
                    
                                
                                #include <iostream>
                                using namespace std;

                                int sum(int a, int b) {
                                    return a + b;
                                }

                                double sum(double a, double b) {
                                    return a + b;
                                }

                                int main() {
                                    cout << "Sum of integers: " << sum(5, 5) << endl;
                                    cout << "Sum of doubles: " << sum(5.2, 5.3) << endl;
                                    return 0;
                                }

    2. Recursive Function

  2. Write a C++ program to calculate the factorial of a number using a recursive function. 
                            Sample Output
                        
                            Enter a number: 5
                            Factorial of 5 is: 120
                        
                    
                                
                                #include <iostream>
                                using namespace std;

                                int factorial(int n) {
                                    if (n == 0) return 1;
                                    else return n * factorial(n - 1);
                                }

                                int main() {
                                    int num;
                                    cout << "Enter a number: ";
                                    cin >> num;
                                    cout << "Factorial of " << num << " is: " << factorial(num) << endl;
                                    return 0;
                                }

    3. Function Pointers

  3. Write a C++ program to use function pointers to select different functions for addition, subtraction, multiplication, and division. 
                            Sample Output
                        
                            Addition: 5 + 3 = 8
                            Subtraction: 5 - 3 = 2
                            Multiplication: 5 * 3 = 15
                            Division: 6 / 3 = 2
                        
                    
                                
                                #include <iostream>
                                using namespace std;

                                int add(int a, int b) { return a + b; }
                                int subtract(int a, int b) { return a - b; }
                                int multiply(int a, int b) { return a * b; }
                                int divide(int a, int b) { return a / b; }

                                int main() {
                                    int (*funcPtr)(int, int);
                                    int a = 5, b = 3;
                                    
                                    funcPtr = add;
                                    cout << "Addition: " << a << " + " << b << " = " << funcPtr(a, b) << endl;
                                    
                                    funcPtr = subtract;
                                    cout << "Subtraction: " << a << " - " << b << " = " << funcPtr(a, b) << endl;
                                    
                                    funcPtr = multiply;
                                    cout << "Multiplication: " << a << " * " << b << " = " << funcPtr(a, b) << endl;
                                    
                                    int c = 6;
                                    funcPtr = divide;
                                    cout << "Division: " << c << " / " << b << " = " << funcPtr(c, b) << endl;
                                    
                                    return 0;
                                }

    4. Pass by Reference

  4. Write a C++ program to demonstrate pass by reference using a function that swaps two variables. 
                            Sample Output
                        
                            Before swap: a = 10, b = 20
                            After swap: a = 20, b = 10
                        
                    
                                
                                #include <iostream>
                                using namespace std;

                                void swap(int &x, int &y) {
                                    int temp = x;
                                    x = y;
                                    y = temp;
                                }

                                int main() {
                                    int a = 10, b = 20;
                                    cout << "Before swap: a = " << a << ", b = " << b << endl;
                                    swap(a, b);
                                    cout << "After swap: a = " << a << ", b = " << b << endl;
                                    return 0;
                                }

    5. Default Arguments

  5. Write a C++ program to demonstrate the use of default arguments in a function. 
                            Sample Output
                        
                            Area of rectangle with length 10 and width 5 is 50
                            Area of rectangle with length 10 and default width is 30
                        
                    
                                
                                #include <iostream>
                                using namespace std;

                                int area(int length, int width = 3) {
                                    return length * width;
                                }

                                int main() {
                                    cout << "Area of rectangle with length 10 and width 5 is " << area(10, 5) << endl;
                                    cout << "Area of rectangle with length 10 and default width is " << area(10) << endl;
                                    return 0;
                                }
