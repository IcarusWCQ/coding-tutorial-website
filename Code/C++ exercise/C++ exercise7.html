Array Rotation

  1. Write a C++ program to rotate an array to the right by a given number of steps. 
                            Sample Input
                        
                            Array: [1, 2, 3, 4, 5]
                            Steps: 2
                        
                    
                                
                                #include <iostream>
                                #include <vector>
                                using namespace std;

                                void rotateArray(vector<int>& arr, int steps) {
                                    int n = arr.size();
                                    steps = steps % n;
                                    reverse(arr.begin(), arr.end());
                                    reverse(arr.begin(), arr.begin() + steps);
                                    reverse(arr.begin() + steps, arr.end());
                                }

                                int main() {
                                    vector<int> arr = {1, 2, 3, 4, 5};
                                    int steps = 2;
                                    rotateArray(arr, steps);
                                    cout << "Rotated Array: ";
                                    for (int num : arr) {
                                        cout << num << " ";
                                    }
                                    cout << endl;
                                    return 0;
                                }

    2. Maximum Subarray Sum (Kadane's Algorithm)

  2. Write a C++ program to find the contiguous subarray within a one-dimensional array of numbers which has the largest sum. 
                            Sample Input
                        
                            Array: [-2, 1, -3, 4, -1, 2, 1, -5, 4]
                        
                    
                                
                                #include <iostream>
                                #include <vector>
                                using namespace std;

                                int maxSubArraySum(vector<int>& arr) {
                                    int max_so_far = arr[0], max_ending_here = arr[0];
                                    for (size_t i = 1; i < arr.size(); i++) {
                                        max_ending_here = max(arr[i], max_ending_here + arr[i]);
                                        max_so_far = max(max_so_far, max_ending_here);
                                    }
                                    return max_so_far;
                                }

                                int main() {
                                    vector<int> arr = {-2, 1, -3, 4, -1, 2, 1, -5, 4};
                                    cout << "Maximum Sum: " << maxSubArraySum(arr) << endl;
                                    return 0;
                                }

    3. Spiral Order Matrix

  3. Write a C++ program to print the elements of a 2D array in spiral order. 
                            Sample Input
                        
                            Matrix:
                            1  2  3
                            4  5  6
                            7  8  9
                        
                    
                                
                                #include <iostream>
                                #include <vector>
                                using namespace std;

                                void spiralOrder(vector<vector<int>>& matrix) {
                                    int m = matrix.size();
                                    if (m == 0) return;
                                    int n = matrix[0].size();
                                    int top = 0, bottom = m - 1, left = 0, right = n - 1;
                                    while (top <= bottom && left <= right) {
                                        for (int i = left; i <= right; i++) cout << matrix[top][i] << " ";
                                        top++;
                                        for (int i = top; i <= bottom; i++) cout << matrix[i][right] << " ";
                                        right--;
                                        if (top <= bottom) {
                                            for (int i = right; i >= left; i--) cout << matrix[bottom][i] << " ";
                                            bottom--;
                                        }
                                        if (left <= right) {
                                            for (int i = bottom; i >= top; i--) cout << matrix[i][left] << " ";
                                            left++;
                                        }
                                    }
                                    cout << endl;
                                }

                                int main() {
                                    vector<vector<int>> matrix = {
                                        {1, 2, 3},
                                        {4, 5, 6},
                                        {7, 8, 9}
                                    };
                                    cout << "Spiral Order: ";
                                    spiralOrder(matrix);
                                    return 0;
                                }

    4. Longest Increasing Subsequence

  4. Write a C++ program to find the length of the longest increasing subsequence in an array. 
                            Sample Input
                        
                            Array: [10, 9, 2, 5, 3, 7, 101, 18]
                        
                    
                                
                                #include <iostream>
                                #include <vector>
                                #include <algorithm>
                                using namespace std;

                                int lengthOfLIS(vector<int>& nums) {
                                    if (nums.empty()) return 0;
                                    vector<int> lis(nums.size(), 1);
                                    for (size_t i = 1; i < nums.size(); i++) {
                                        for (size_t j = 0; j < i; j++) {
                                            if (nums[i] > nums[j] && lis[i] < lis[j] + 1) {
                                                lis[i] = lis[j] + 1;
                                            }
                                        }
                                    }
                                    return *max_element(lis.begin(), lis.end());
                                }

                                int main() {
                                    vector<int> nums = {10, 9, 2, 5, 3, 7, 101, 18};
                                    cout << "Length of Longest Increasing Subsequence: " << lengthOfLIS(nums) << endl;
                                    return 0;
                                }

    5. Matrix Multiplication

  5. Write a C++ program to multiply two matrices and print the resulting matrix. 
                            Sample Input
                        
                            Matrix A:
                            1  2
                            3  4

                            Matrix B:
                            5  6
                            7  8
                        
                    
                                
                                #include <iostream>
                                #include <vector>
                                using namespace std;

                                void multiplyMatrices(vector<vector<int>>& A, vector<vector<int>>& B, vector<vector<int>>& C) {
                                    int n = A.size();
                                    int m = A[0].size();
                                    int p = B[0].size();
                                    for (int i = 0; i < n; i++) {
                                        for (int j = 0; j < p; j++) {
                                            C[i][j] = 0;
                                            for (int k = 0; k < m; k++) {
                                                C[i][j] += A[i][k] * B[k][j];
                                            }
                                        }
                                    }
                                }

                                int main() {
                                    vector<vector<int>> A = {{1, 2}, {3, 4}};
                                    vector<vector<int>> B = {{5, 6}, {7, 8}};
                                    vector<vector<int>> C(2, vector<int>(2));

                                    multiplyMatrices(A, B, C);

                                    cout << "Resultant Matrix:" << endl;
                                    for (int i = 0; i < 2; i++) {
                                        for (int j = 0; j < 2; j++) {
                                            cout << C[i][j] << " ";
                                        }
                                        cout << endl;
                                    }
                                    return 0;
                                }
