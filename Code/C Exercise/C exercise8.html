Function

  1. Write a program in C to show the simple structure of a function. 
                        Sample Output
                    
                        The total is :  11  
                    
                
                                
                                #include <stdio.h>
                                int sum (int, int);//function declaration
                                int main (void)
                                {
                                    int total;
                                    printf("\n\n Function : a simple structure of function :\n");
                                    printf("------------------------------------------------\n");	
                                    total = sum (5, 6);//function call
                                    printf ("The total is :  %d\n", total);
                                    return 0;
                                }
                                
                                int sum (int a, int b) //function definition
                                {
                                    int s;
                                    s=a+b;
                                    return s; //function returning a value
                                }
  2. Write a program in C to find the square of any number using the function. 
                        Sample Output
                    
                        Test Data :
                        Input any number for square : 20

                        The square of 20 is : 400.00
                    
                
                                
                                #include <stdio.h> // Including the input-output stream header file
                                double square(double num)
                                {
                                    return (num * num);
                                }
                                int main()
                                {
                                    int num;
                                    double n;
                                    printf("\n\n Function : find square of any number :\n");
                                    printf("------------------------------------------------\n");	
                                    
                                    printf("Input any number for square : ");
                                    scanf("%d", &num);
                                    n = square(num);
                                    printf("The square of %d is : %.2f\n", num, n); 
                                    return 0;
                                }
