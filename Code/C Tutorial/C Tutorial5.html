  3. Write a program that calculates the factorial of a number using a do-while loop.
    4.                     
#include <stdio.h>

int main() {
    int num = 5;
    int factorial = 1;
    int i = 1;

    do {
        factorial *= i;
        i++;
    } while (i <= num);

    printf("Factorial of %d is %d\n", num, factorial);

    return 0;
}
                    
                
                        Sample Output:
                    
Factorial of 5 is 120

int main() {

  5. Write a program that reverses a number using a while loop.
    6.                     
#include <stdio.h>

int main() {
    int num = 12345;
    int reversed = 0;

    while (num != 0) {
        int digit = num % 10;
        reversed = reversed * 10 + digit;
        num /= 10;
    }

    printf("Reversed number is %d\n", reversed);

    return 0;
}
                    
                
                        Sample Output:
                    
Reversed number is 54321
