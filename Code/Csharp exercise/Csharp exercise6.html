Logical Operators in C#

AND (&&) Operator

  1. Write a C# program to check if a number is between 10 and 20 (inclusive) using the AND operator. 
                            Sample Output
                        
                            Enter a number: 15
                            The number is between 10 and 20.
                        
                    
                                
                                using System;

                                class Program
                                {
                                    static void Main()
                                    {
                                        Console.Write("Enter a number: ");
                                        int num = Convert.ToInt32(Console.ReadLine());

                                        if (num >= 10 && num <= 20)
                                        {
                                            Console.WriteLine("The number is between 10 and 20.");
                                        }
                                        else
                                        {
                                            Console.WriteLine("The number is not between 10 and 20.");
                                        }
                                    }
                                }
  2. Write a C# program to check if a person is eligible to vote (age >= 18 and is a citizen) using the AND operator. 
                            Sample Output
                        
                            Enter age: 20
                            Are you a citizen (yes/no): yes
                            You are eligible to vote.
                        
                    
                                
                                using System;

                                class Program
                                {
                                    static void Main()
                                    {
                                        Console.Write("Enter age: ");
                                        int age = Convert.ToInt32(Console.ReadLine());

                                        Console.Write("Are you a citizen (yes/no): ");
                                        string citizen = Console.ReadLine();

                                        if (age >= 18 && citizen == "yes")
                                        {
                                            Console.WriteLine("You are eligible to vote.");
                                        }
                                        else
                                        {
                                            Console.WriteLine("You are not eligible to vote.");
                                        }
                                    }
                                }
  3. Write a C# program to check if a character is an uppercase letter and a vowel using the AND operator. 
                            Sample Output
                        
                            Enter a character: A
                            A is an uppercase vowel.
                        
                    
                                
                                using System;

                                class Program
                                {
                                    static void Main()
                                    {
                                        Console.Write("Enter a character: ");
                                        char ch = Convert.ToChar(Console.ReadLine().ToUpper());

                                        if ((ch == 'A' || ch == 'E' || ch == 'I' || ch == 'O' || ch == 'U') && char.IsUpper(ch))
                                        {
                                            Console.WriteLine($"{ch} is an uppercase vowel.");
                                        }
                                        else
                                        {
                                            Console.WriteLine($"{ch} is not an uppercase vowel.");
                                        }
                                    }
                                }

    4. OR (||) Operator

  4. Write a C# program to check if a number is less than 10 or greater than 20 using the OR operator. 
                            Sample Output
                        
                            Enter a number: 25
                            The number is less than 10 or greater than 20.
                        
                    
                                
                                using System;

                                class Program
                                {
                                    static void Main()
                                    {
                                        Console.Write("Enter a number: ");
                                        int num = Convert.ToInt32(Console.ReadLine());

                                        if (num < 10 || num > 20)
                                        {
                                            Console.WriteLine("The number is less than 10 or greater than 20.");
                                        }
                                        else
                                        {
                                            Console.WriteLine("The number is between 10 and 20.");
                                        }
                                    }
                                }

    7. NOT (!) Operator

  7. Write a C# program to check if a number is not equal to 10 using the NOT operator. 
                            Sample Output
                        
                            Enter a number: 15
                            The number is not equal to 10.
                        
                    
                                
                                using System;

                                class Program
                                {
                                    static void Main()
                                    {
                                        Console.Write("Enter a number: ");
                                        int num = Convert.ToInt32(Console.ReadLine());

                                        if (num != 10)
                                        {
                                            Console.WriteLine("The number is not equal to 10.");
                                        }
                                        else
                                        {
                                            Console.WriteLine("The number is equal to 10.");
                                        }
                                    }
                                }
  8. Write a C# program to check if a character is not a vowel (a, e, i, o, u) using the NOT operator. 
                            Sample Output
                        
                            Enter a character: k
                            The character is not a vowel.
                        
                    
                                
                                using System;

                                class Program
                                {
                                    static void Main()
                                    {
                                        Console.Write("Enter a character: ");
                                        char ch = Convert.ToChar(Console.ReadLine().ToLower());

                                        if (!(ch == 'a' || ch == 'e' || ch == 'i' || ch == 'o' || ch == 'u'))
                                        {
                                            Console.WriteLine("The character is not a vowel.");
                                        }
                                        else
                                        {
                                            Console.WriteLine("The character is a vowel.");
                                        }
                                    }
                                }
  9. Write a C# program to check if a person is not eligible to vote (age < 18) using the NOT operator. 
                            Sample Output
                        
                            Enter age: 16
                            You are not eligible to vote.
                        
                    
                                
                                using System;

                                class Program
                                {
                                    static void Main()
                                    {
                                        Console.Write("Enter age: ");
                                        int age = Convert.ToInt32(Console.ReadLine());

                                        if (!(age >= 18))
                                        {
                                            Console.WriteLine("You are not eligible to vote.");
                                        }
                                        else
                                        {
                                            Console.WriteLine("You are eligible to vote.");
                                        }
                                    }
                                }
