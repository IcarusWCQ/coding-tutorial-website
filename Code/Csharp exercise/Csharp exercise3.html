Switch Case in C#

Basic Calculator

  1. Write a C# program to create a simple calculator using switch-case that performs addition, subtraction, multiplication, and division. 
                            Sample Output
                        
                            Enter first number: 10
                            Enter second number: 5
                            Enter operator (+, -, *, /): *
                            Result: 50
                        
                    
                                
                                using System;

                                class Program
                                {
                                    static void Main()
                                    {
                                        Console.Write("Enter first number: ");
                                        double num1 = Convert.ToDouble(Console.ReadLine());
                                        Console.Write("Enter second number: ");
                                        double num2 = Convert.ToDouble(Console.ReadLine());
                                        Console.Write("Enter operator (+, -, *, /): ");
                                        char op = Console.ReadKey().KeyChar;
                                        Console.WriteLine();

                                        double result = 0;
                                        bool validOperation = true;

                                        switch (op)
                                        {
                                            case '+':
                                                result = num1 + num2;
                                                break;
                                            case '-':
                                                result = num1 - num2;
                                                break;
                                            case '*':
                                                result = num1 * num2;
                                                break;
                                            case '/':
                                                if (num2 != 0)
                                                {
                                                    result = num1 / num2;
                                                }
                                                else
                                                {
                                                    Console.WriteLine("Division by zero is not allowed.");
                                                    validOperation = false;
                                                }
                                                break;
                                            default:
                                                Console.WriteLine("Invalid operator.");
                                                validOperation = false;
                                                break;
                                        }

                                        if (validOperation)
                                        {
                                            Console.WriteLine("Result: " + result);
                                        }
                                    }
                                }

    2. Day of the Week

  2. Write a C# program to print the day of the week based on a number input (1 for Sunday, 2 for Monday, ..., 7 for Saturday) using switch-case. 
                            Sample Output
                        
                            Enter a number (1-7): 3
                            Day of the week: Tuesday
                        
                    
                                
                                using System;

                                class Program
                                {
                                    static void Main()
                                    {
                                        Console.Write("Enter a number (1-7): ");
                                        int day = Convert.ToInt32(Console.ReadLine());

                                        switch (day)
                                        {
                                            case 1:
                                                Console.WriteLine("Day of the week: Sunday");
                                                break;
                                            case 2:
                                                Console.WriteLine("Day of the week: Monday");
                                                break;
                                            case 3:
                                                Console.WriteLine("Day of the week: Tuesday");
                                                break;
                                            case 4:
                                                Console.WriteLine("Day of the week: Wednesday");
                                                break;
                                            case 5:
                                                Console.WriteLine("Day of the week: Thursday");
                                                break;
                                            case 6:
                                                Console.WriteLine("Day of the week: Friday");
                                                break;
                                            case 7:
                                                Console.WriteLine("Day of the week: Saturday");
                                                break;
                                            default:
                                                Console.WriteLine("Invalid input. Please enter a number between 1 and 7.");
                                                break;
                                        }
                                    }
                                }

    3. Grade Evaluation

  3. Write a C# program to evaluate the grade of a student based on their score using switch-case. The grades are: A (90-100), B (80-89), C (70-79), D (60-69), F (0-59). 
                            Sample Output
                        
                            Enter the score (0-100): 85
                            Grade: B
                        
                    
                                
                                using System;

                                class Program
                                {
                                    static void Main()
                                    {
                                        Console.Write("Enter the score (0-100): ");
                                        int score = Convert.ToInt32(Console.ReadLine());

                                        switch (score)
                                        {
                                            case int n when (n >= 90 && n <= 100):
                                                Console.WriteLine("Grade: A");
                                                break;
                                            case int n when (n >= 80 && n < 90):
                                                Console.WriteLine("Grade: B");
                                                break;
                                            case int n when (n >= 70 && n < 80):
                                                Console.WriteLine("Grade: C");
                                                break;
                                            case int n when (n >= 60 && n < 70):
                                                Console.WriteLine("Grade: D");
                                                break;
                                            case int n when (n >= 0 && n < 60):
                                                Console.WriteLine("Grade: F");
                                                break;
                                            default:
                                                Console.WriteLine("Invalid score. Please enter a number between 0 and 100.");
                                                break;
                                        }
                                    }
                                }

    2. Simple Vending Machine

  2. Write a C# program to simulate a simple vending machine. The user can select from three products: Soda ($1), Chips ($0.5), and Candy ($0.75). Use switch-case to handle the product selection and display the price. 
                            Sample Output
                        
                            Select a product (1 - Soda, 2 - Chips, 3 - Candy): 2
                            You selected Chips. The price is $0.5
                        
                    
                                
                                using System;

                                class Program
                                {
                                    static void Main()
                                    {
                                        Console.Write("Select a product (1 - Soda, 2 - Chips, 3 - Candy): ");
                                        int product = Convert.ToInt32(Console.ReadLine());

                                        switch (product)
                                        {
                                            case 1:
                                                Console.WriteLine("You selected Soda. The price is $1.");
                                                break;
                                            case 2:
                                                Console.WriteLine("You selected Chips. The price is $0.5.");
                                                break;
                                            case 3:
                                                Console.WriteLine("You selected Candy. The price is $0.75.");
                                                break;
                                            default:
                                                Console.WriteLine("Invalid selection. Please choose 1, 2, or 3.");
                                                break;
                                        }
                                    }
                                }
