Arrays in C#

Find the Maximum Element

  1. Write a C# program to find the maximum element in an array of integers. 
                            Sample Output
                        
                            Enter the number of elements: 5
                            Enter element 1: 3
                            Enter element 2: 1
                            Enter element 3: 4
                            Enter element 4: 1
                            Enter element 5: 5
                            The maximum element is 5.
                        
                    
                                
                                using System;

                                class Program
                                {
                                    static void Main()
                                    {
                                        Console.Write("Enter the number of elements: ");
                                        int n = Convert.ToInt32(Console.ReadLine());

                                        int[] arr = new int[n];
                                        for (int i = 0; i < n; i++)
                                        {
                                            Console.Write($"Enter element {i + 1}: ");
                                            arr[i] = Convert.ToInt32(Console.ReadLine());
                                        }

                                        int max = arr[0];
                                        for (int i = 1; i < arr.Length; i++)
                                        {
                                            if (arr[i] > max)
                                            {
                                                max = arr[i];
                                            }
                                        }

                                        Console.WriteLine($"The maximum element is {max}.");
                                    }
                                }

    2. Reverse an Array

  2. Write a C# program to reverse the elements of an array. 
                            Sample Output
                        
                            Enter the number of elements: 5
                            Enter element 1: 1
                            Enter element 2: 2
                            Enter element 3: 3
                            Enter element 4: 4
                            Enter element 5: 5
                            Reversed array: 5 4 3 2 1
                        
                    
                                
                                using System;

                                class Program
                                {
                                    static void Main()
                                    {
                                        Console.Write("Enter the number of elements: ");
                                        int n = Convert.ToInt32(Console.ReadLine());

                                        int[] arr = new int[n];
                                        for (int i = 0; i < n; i++)
                                        {
                                            Console.Write($"Enter element {i + 1}: ");
                                            arr[i] = Convert.ToInt32(Console.ReadLine());
                                        }

                                        Array.Reverse(arr);

                                        Console.Write("Reversed array: ");
                                        foreach (int elem in arr)
                                        {
                                            Console.Write(elem + " ");
                                        }
                                    }
                                }

    3. Sum of Elements

  3. Write a C# program to find the sum of all elements in an array. 
                            Sample Output

                            Enter element 1: 1
                            Enter element 2: 2
                            Enter element 3: 3
                            Enter element 4: 4
                            Enter element 5: 5

                                using System;

                                class Program
                                {
                                    static void Main()
                                    {

    4. Find Duplicate Elements

  4. Write a C# program to find duplicate elements in an array. 
                            Sample Output
                        
                            Enter the number of elements: 6
                            Enter element 1: 1
                            Enter element 2: 2
                            Enter element 3: 3
                            Enter element 4: 2
                            Enter element 5: 4
                            Enter element 6: 1
                            Duplicate elements: 1 2
                        
                    
                                
                                using System;
                                using System.Collections.Generic;

                                class Program
                                {
                                    static void Main()
                                    {
                                        Console.Write("Enter the number of elements: ");
                                        int n = Convert.ToInt32(Console.ReadLine());

                                        int[] arr = new int[n];
                                        for (int i = 0; i < n; i++)
                                        {
                                            Console.Write($"Enter element {i + 1}: ");
                                            arr[i] = Convert.ToInt32(Console.ReadLine());
                                        }

                                        HashSet uniqueElements = new HashSet();
                                        HashSet duplicateElements = new HashSet();

                                        foreach (int elem in arr)
                                        {
                                            if (!uniqueElements.Add(elem))
                                            {
                                                duplicateElements.Add(elem);
                                            }
                                        }

                                        Console.Write("Duplicate elements: ");
                                        foreach (int elem in duplicateElements)
                                        {
                                            Console.Write(elem + " ");
                                        }
                                    }
                                }

    5. Merge Two Arrays

  5. Write a C# program to merge two arrays into a third array. 
                            Sample Output
                        
                            Enter the number of elements in the first array: 3
                            Enter element 1: 1
                            Enter element 2: 2
                            Enter element 3: 3
                            Enter the number of elements in the second array: 3
                            Enter element 1: 4
                            Enter element 2: 5
                            Enter element 3: 6
                            Merged array: 1 2 3 4 5 6
                        
                    
                                
                                using System;

                                class Program
                                {
                                    static void Main()
                                    {
                                        Console.Write("Enter the number of elements in the first array: ");
                                        int n1 = Convert.ToInt32(Console.ReadLine());

                                        int[] arr1 = new int[n1];
                                        for (int i = 0; i < n1; i++)
                                        {
                                            Console.Write($"Enter element {i + 1}: ");
                                            arr1[i] = Convert.ToInt32(Console.ReadLine());
                                        }

                                        Console.Write("Enter the number of elements in the second array: ");
                                        int n2 = Convert.ToInt32(Console.ReadLine());

                                        int[] arr2 = new int[n2];
                                        for (int i = 0; i < n2; i++)
                                        {
                                            Console.Write($"Enter element {i + 1}: ");
                                            arr2[i] = Convert.ToInt32(Console.ReadLine());
                                        }

                                        int[] mergedArray = new int[n1 + n2];
                                        arr1.CopyTo(mergedArray, 0);
                                        arr2.CopyTo(mergedArray, n1);

                                        Console.Write("Merged array: ");
                                        foreach (int elem in mergedArray)
                                        {
                                            Console.Write(elem + " ");
                                        }
                                    }
                                }
