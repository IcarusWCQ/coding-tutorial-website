Basic Algorithms in C#

1. Bubble Sort

  1. Write a C# program to implement the Bubble Sort algorithm to sort an array of integers. 
                            Sample Output
                        
                            Original array: 5, 3, 8, 4, 2
                            Sorted array: 2, 3, 4, 5, 8
                        
                    
                                
                                using System;

                                class Program
                                {
                                    static void Main()
                                    {
                                        int[] arr = { 5, 3, 8, 4, 2 };
                                        Console.WriteLine("Original array: " + string.Join(", ", arr));

                                        for (int i = 0; i < arr.Length - 1; i++)
                                        {
                                            for (int j = 0; j < arr.Length - i - 1; j++)
                                            {
                                                if (arr[j] > arr[j + 1])
                                                {
                                                    int temp = arr[j];
                                                    arr[j] = arr[j + 1];
                                                    arr[j + 1] = temp;
                                                }
                                            }
                                        }

                                        Console.WriteLine("Sorted array: " + string.Join(", ", arr));
                                    }
                                }

    2. Binary Search

  2. Write a C# program to implement the Binary Search algorithm to find a specific element in a sorted array of integers. 
                            Sample Output
                        
                            Sorted array: 1, 2, 3, 4, 5
                            Enter the number to search: 4
                            4 found at index 3
                        
                    
                                
                                using System;

                                class Program
                                {
                                    static void Main()
                                    {
                                        int[] arr = { 1, 2, 3, 4, 5 };
                                        Console.WriteLine("Sorted array: " + string.Join(", ", arr));

                                        Console.Write("Enter the number to search: ");
                                        int target = Convert.ToInt32(Console.ReadLine());

                                        int result = BinarySearch(arr, target);

                                        if (result != -1)
                                        {
                                            Console.WriteLine($"{target} found at index {result}");
                                        }
                                        else
                                        {
                                            Console.WriteLine($"{target} not found in the array.");
                                        }
                                    }

                                    static int BinarySearch(int[] arr, int target)
                                    {
                                        int left = 0;
                                        int right = arr.Length - 1;

                                        while (left <= right)
                                        {
                                            int mid = left + (right - left) / 2;

                                            if (arr[mid] == target)
                                            {
                                                return mid;
                                            }

                                            if (arr[mid] < target)
                                            {
                                                left = mid + 1;
                                            }
                                            else
                                            {
                                                right = mid - 1;
                                            }
                                        }

                                        return -1;
                                    }
                                }

    3. Find the GCD

  3. Write a C# program to find the Greatest Common Divisor (GCD) of two numbers using the Euclidean algorithm. 
                            Sample Output
                        
                            Enter first number: 56
                            Enter second number: 98
                            GCD of 56 and 98 is 14
                        
                    
                                
                                using System;

                                class Program
                                {
                                    static void Main()
                                    {
                                        Console.Write("Enter first number: ");
                                        int a = Convert.ToInt32(Console.ReadLine());
                                        Console.Write("Enter second number: ");
                                        int b = Convert.ToInt32(Console.ReadLine());

                                        int gcd = GCD(a, b);
                                        Console.WriteLine($"GCD of {a} and {b} is {gcd}");
                                    }

                                    static int GCD(int a, int b)
                                    {
                                        while (b != 0)
                                        {
                                            int temp = b;
                                            b = a % b;
                                            a = temp;
                                        }
                                        return a;
                                    }
                                }

    4. 4. Fibonacci Sequence

  4. Write a C# program to generate the first n Fibonacci numbers. 
                            Sample Output
                        
                            Enter the number of terms: 5
                            Fibonacci Sequence: 0, 1, 1, 2, 3
                        
                    
                                
                                using System;

                                class Program
                                {
                                    static void Main()
                                    {
                                        Console.Write("Enter the number of terms: ");
                                        int n = Convert.ToInt32(Console.ReadLine());

                                        int[] fibonacci = new int[n];
                                        if (n > 0)
                                            fibonacci[0] = 0;
                                        if (n > 1)
                                            fibonacci[1] = 1;

                                        for (int i = 2; i < n; i++)
                                        {
                                            fibonacci[i] = fibonacci[i - 1] + fibonacci[i - 2];
                                        }

                                        Console.WriteLine("Fibonacci Sequence: " + string.Join(", ", fibonacci));
                                    }
                                }

    5. Merge Sort

  5. Write a C# program to implement the Merge Sort algorithm to sort an array of integers. 
                            Sample Output
                        
                            Original array: 12, 11, 13, 5, 6, 7
                            Sorted array: 5, 6, 7, 11, 12, 13
                        
                    
                                
                                using System;

                                class Program
                                {
                                    static void Main()
                                    {
                                        int[] arr = { 12, 11, 13, 5, 6, 7 };
                                        Console.WriteLine("Original array: " + string.Join(", ", arr));

                                        MergeSort(arr, 0, arr.Length - 1);

                                        Console.WriteLine("Sorted array: " + string.Join(", ", arr));
                                    }

                                    static void MergeSort(int[] arr, int left, int right)
                                    {
                                        if (left < right)
                                        {
                                            int mid = left + (right - left) / 2;

                                            MergeSort(arr, left, mid);
                                            MergeSort(arr, mid + 1, right);

                                            Merge(arr, left, mid, right);
                                        }
                                    }

                                    static void Merge(int[] arr, int left, int mid, int right)
                                    {
                                        int n1 = mid - left + 1;
                                        int n2 = right - mid;

                                        int[] L = new int[n1];
                                        int[] R = new int[n2];

                                        Array.Copy(arr, left, L, 0, n1);
                                        Array.Copy(arr, mid + 1, R, 0, n2);

                                        int i = 0, j = 0, k = left;
                                        while (i < n1 && j < n2)
                                        {
                                            if (L[i] <= R[j])
                                            {
                                                arr[k] = L[i];
                                                i++;
                                            }
                                            else
                                            {
                                                arr[k] = R[j];
                                                j++;
                                            }
                                            k++;
                                        }

                                        while (i < n1)
                                        {
                                            arr[k] = L[i];
                                            i++;
                                            k++;
                                        }

                                        while (j < n2)
                                        {
                                            arr[k] = R[j];
                                            j++;
                                            k++;
                                        }
                                    }
                                }
