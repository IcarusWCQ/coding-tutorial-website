  1. Write a program in C++ to check whether a number is even or odd. 
                        Sample Output
                    
                        Input a number: 5
                        5 is an odd number.
                    
                
                                
                                #include <iostream> // Including the input-output stream header file
                                using namespace std; // Using the standard namespace to avoid writing std::

                                int main() // Start of the main function
                                {
                                    int num; // Declare an integer variable 'num'
                                    
                                    // Prompt the user to input a number
                                    cout << "Input a number: ";
                                    cin >> num; // Read the value entered by the user
                                    
                                    // Check whether the number is even or odd
                                    if (num % 2 == 0)
                                    {
                                        cout << num << " is an even number." << endl; // Output if the number is even
                                    }
                                    else
                                    {
                                        cout << num << " is an odd number." << endl; // Output if the number is odd
                                    }
                                    
                                    return 0; // Indicating successful completion of the program
                                }

  3. Write a program in C++ to find the largest of three numbers. 
                        Sample Output
                    
                        Input the first number: 25
                        Input the second number: 78
                        Input the third number: 87
                        The largest number is 87
                    
                
                                
                                #include <iostream> // Including the input-output stream header file
                                using namespace std; // Using the standard namespace to avoid writing std::

                                int main() // Start of the main function
                                {
                                    int num1, num2, num3; // Declare integer variables 'num1', 'num2', and 'num3'
                                    
                                    // Prompt the user to input the first number
                                    cout << "Input the first number: ";
                                    cin >> num1; // Read the value entered by the user
                                    
                                    // Prompt the user to input the second number
                                    cout << "Input the second number: ";
                                    cin >> num2; // Read the value entered by the user
                                    
                                    // Prompt the user to input the third number
                                    cout << "Input the third number: ";
                                    cin >> num3; // Read the value entered by the user
                                    
                                    // Find the largest number among the three
                                    if (num1 >= num2 && num1 >= num3)
                                    {
                                        cout << "The largest number is " << num1 << endl; // Output if 'num1' is the largest
                                    }
                                    else if (num2 >= num1 && num2 >= num3)
                                    {
                                        cout << "The largest number is " << num2 << endl; // Output if 'num2' is the largest
                                    }
                                    else
                                    {
                                        cout << "The largest number is " << num3 << endl; // Output if 'num3' is the largest
                                    }
                                    
                                    return 0; // Indicating successful completion of the program
                                }
  4. Write a program in C++ to check whether a character is a vowel or consonant. 
                        Sample Output

                                }
  5. Write a program in C++ to check whether a year is a leap year or not. 
                        Sample Output
                    
                        Input a year: 2020
                        2020 is a leap year.
                    
                
                                
                                #include <iostream> // Including the input-output stream header file
                                using namespace std; // Using the standard namespace to avoid writing std::

                                int main() // Start of the main function
                                {
                                    int year; // Declare an integer variable 'year'
                                    
                                    // Prompt the user to input a year
                                    cout << "Input a year: ";
                                    cin >> year; // Read the value entered by the user
                                    
                                    // Check whether the year is a leap year or not
                                    if ((year % 4 == 0 && year % 100 != 0) || (year % 400 == 0))
                                    {
                                        cout << year << " is a leap year." << endl; // Output if the year is a leap year
                                    }
                                    else
                                    {
                                        cout << year << " is not a leap year." << endl; // Output if the year is not a leap year
                                    }
                                    
                                    return 0; // Indicating successful completion of the program
                                }
