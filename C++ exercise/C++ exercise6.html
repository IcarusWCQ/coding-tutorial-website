Logical Operators

AND (&&)

  1. Write a program in C++ to check if a number is between 10 and 20 (inclusive) using the AND operator. 
                            Sample Output
                        
                            Input a number: 15
                            The number is between 10 and 20.
                        
                    
                                
                                #include <iostream>
                                using namespace std;

                                int main() {
                                    int num;
                                    cout << "Input a number: ";
                                    cin >> num;

                                    if (num >= 10 && num <= 20) {
                                        cout << "The number is between 10 and 20." << endl;
                                    } else {
                                        cout << "The number is not between 10 and 20." << endl;
                                    }

                                    return 0;
                                }
  2. Write a program in C++ to check if a character is a vowel (a, e, i, o, u) using the AND operator. 
                            Sample Output
                        
                            Input a character: e
                            The character is a vowel.
                        
                    
                                
                                #include <iostream>
                                using namespace std;

                                int main() {
                                    char ch;
                                    cout << "Input a character: ";
                                    cin >> ch;

                                    if (ch == 'a' && ch == 'e' && ch == 'i' && ch == 'o' && ch == 'u') {
                                        cout << "The character is a vowel." << endl;
                                    } else {
                                        cout << "The character is not a vowel." << endl;
                                    }

                                    return 0;
                                }
  3. Write a program in C++ to check if a person is eligible to vote (age >= 18 and is a citizen) using the AND operator. 
                            Sample Output
                        
                            Input age: 20
                            Are you a citizen (1 for yes, 0 for no): 1
                            You are eligible to vote.
                        
                    
                                
                                #include <iostream>
                                using namespace std;

                                int main() {
                                    int age, citizen;
                                    cout << "Input age: ";
                                    cin >> age;
                                    cout << "Are you a citizen (1 for yes, 0 for no): ";
                                    cin >> citizen;

                                    if (age >= 18 && citizen == 1) {
                                        cout << "You are eligible to vote." << endl;
                                    } else {
                                        cout << "You are not eligible to vote." << endl;
                                    }

                                    return 0;
                                }

NOT (!)

  1. Write a program in C++ to check if a number is not equal to 10 using the NOT operator. 
                            Sample Output
                        
                            Input a number: 15
                            The number is not equal to 10.
                        
                    
                                
                                #include <iostream>
                                using namespace std;

                                int main() {
                                    int num;
                                    cout << "Input a number: ";
                                    cin >> num;

                                    if (num != 10) {
                                        cout << "The number is not equal to 10." << endl;
                                    } else {
                                        cout << "The number is equal to 10." << endl;
                                    }

                                    return 0;
                                }
  2. Write a program in C++ to check if a character is not a vowel (a, e, i, o, u) using the NOT operator. 
                            Sample Output
                        
                            Input a character: k
                            The character is not a vowel.
                        
                    
                                
                                #include <iostream>
                                using namespace std;

                                int main() {
                                    char ch;
                                    cout << "Input a character: ";
                                    cin >> ch;

                                    if (ch != 'a' && ch != 'e' && ch != 'i' && ch != 'o' && ch != 'u') {
                                        cout << "The character is not a vowel." << endl;
                                    } else {
                                        cout << "The character is a vowel." << endl;
                                    }

                                    return 0;
                                }
  3. Write a program in C++ to check if a person is not eligible to vote (age < 18) using the NOT operator. 
                            Sample Output
                        
                            Input age: 16
                            You are not eligible to vote.
                        
                    
                                
                                #include <iostream>
                                using namespace std;

                                int main() {
                                    int age;
                                    cout << "Input age: ";
                                    cin >> age;

                                    if (!(age >= 18)) {
                                        cout << "You are not eligible to vote." << endl;
                                    } else {
                                        cout << "You are eligible to vote." << endl;
                                    }

                                    return 0;
                                }
