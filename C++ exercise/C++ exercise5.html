Switch Case

  1. Write a program in C++ to display the weekday name based on the day number (1-7). 
                        Sample Output
                    
                        Input a day number: 3
                        Wednesday
                    
                
                                
                                #include <iostream> // Including the input-output stream header file
                                using namespace std; // Using the standard namespace to avoid writing std::

                                int main() // Start of the main function
                                {
                                    int day; // Declare an integer variable 'day'
                                    
                                    // Prompt the user to input a day number
                                    cout << "Input a day number (1-7): ";
                                    cin >> day; // Read the value entered by the user
                                    
                                    // Display the weekday name based on the day number
                                    switch (day)
                                    {
                                        case 1:
                                            cout << "Sunday" << endl;
                                            break;
                                        case 2:
                                            cout << "Monday" << endl;
                                            break;
                                        case 3:
                                            cout << "Tuesday" << endl;
                                            break;
                                        case 4:
                                            cout << "Wednesday" << endl;
                                            break;
                                        case 5:
                                            cout << "Thursday" << endl;
                                            break;
                                        case 6:
                                            cout << "Friday" << endl;
                                            break;
                                        case 7:
                                            cout << "Saturday" << endl;
                                            break;
                                        default:
                                            cout << "Invalid day number" << endl;
                                            break;
                                    }
                                    
                                    return 0; // Indicating successful completion of the program
                                }

  3. Write a program in C++ to perform arithmetic operations based on the user’s choice. 
                        Sample Output
                    
                        Input first number: 10
                        Input second number: 20
                        Input your choice (1: Add, 2: Subtract, 3: Multiply, 4: Divide): 1
                        Result: 30
                    
                
                                
                                #include <iostream> // Including the input-output stream header file
                                using namespace std; // Using the standard namespace to avoid writing std::

                                int main() // Start of the main function
                                {
                                    int num1, num2, choice; // Declare integer variables 'num1', 'num2', and 'choice'
                                    
                                    // Prompt the user to input the first number
                                    cout << "Input first number: ";
                                    cin >> num1; // Read the value entered by the user
                                    
                                    // Prompt the user to input the second number
                                    cout << "Input second number: ";
                                    cin >> num2; // Read the value entered by the user
                                    
                                    // Prompt the user to input their choice for arithmetic operation
                                    cout << "Input your choice (1: Add, 2: Subtract, 3: Multiply, 4: Divide): ";
                                    cin >> choice; // Read the value entered by the user
                                    
                                    // Perform arithmetic operations based on the user’s choice
                                    switch (choice)
                                    {
                                        case 1:
                                            cout << "Result: " << num1 + num2 << endl;
                                            break;
                                        case 2:
                                            cout << "Result: " << num1 - num2 << endl;
                                            break;
                                        case 3:
                                            cout << "Result: " << num1 * num2 << endl;
                                            break;
                                        case 4:
                                            cout << "Result: " << num1 / num2 << endl;
                                            break;
                                        default:
                                            cout << "Invalid choice" << endl;
                                            break;
                                    }
                                    
                                    return 0; // Indicating successful completion of the program
                                }
  4. Write a program in C++ to display a message based on the grade (A-F) entered by the user. 
                        Sample Output
                    
                        Input your grade (A-F): B
                        Good job!
                    
                
                                
                                #include <iostream> // Including the input-output stream header file
                                using namespace std; // Using the standard namespace to avoid writing std::

                                int main() // Start of the main function
                                {
                                    char grade; // Declare a character variable 'grade'
                                    
                                    // Prompt the user to input their grade
                                    cout << "Input your grade (A-F): ";
                                    cin >> grade; // Read the value entered by the user
                                    
                                    // Display a message based on the grade entered by the user
                                    switch (grade)
                                    {
                                        case 'A':
                                            cout << "Excellent!" << endl;
                                            break;
                                        case 'B':
                                            cout << "Good job!" << endl;
                                            break;
                                        case 'C':
                                            cout << "Well done!" << endl;
                                            break;
                                        case 'D':
                                            cout << "You passed." << endl;
                                            break;
                                        case 'F':
                                            cout << "Better try again." << endl;
                                            break;
                                        default:
                                            cout << "Invalid grade" << endl;
                                            break;
                                    }
                                    
                                    return 0; // Indicating successful completion of the program
                                }
