Pointer Arithmetic

  1. Write a C++ program to demonstrate pointer arithmetic by iterating through an array using a pointer. 
                            Sample Output
                        
                            Array: [10, 20, 30, 40, 50]
                            Using pointer arithmetic to iterate through the array:
                            10 20 30 40 50
                        
                    
                                
                                #include <iostream>
                                using namespace std;

                                int main() {
                                    int arr[] = {10, 20, 30, 40, 50};
                                    int *ptr = arr;
                                    cout << "Using pointer arithmetic to iterate through the array:" << endl;
                                    for (int i = 0; i < 5; i++) {
                                        cout << *(ptr + i) << " ";
                                    }
                                    cout << endl;
                                    return 0;
                                }

    2. Swapping Variables

  2. Write a C++ program to swap two variables using pointers. 
                            Sample Output
                        
                            Before swap: a = 10, b = 20
                            After swap: a = 20, b = 10
                        
                    
                                
                                #include <iostream>
                                using namespace std;

                                void swap(int *x, int *y) {
                                    int temp = *x;
                                    *x = *y;
                                    *y = temp;
                                }

                                int main() {
                                    int a = 10, b = 20;
                                    cout << "Before swap: a = " << a << ", b = " << b << endl;
                                    swap(&a, &b);
                                    cout << "After swap: a = " << a << ", b = " << b << endl;
                                    return 0;
                                }

    3. Dynamic Memory Allocation

  3. Write a C++ program to allocate memory dynamically for an array using pointers and then release the memory. 
                            Sample Output
                        
                            Enter number of elements: 5
                            Enter elements: 1 2 3 4 5
                            Elements are: 1 2 3 4 5
                        
                    
                                
                                #include <iostream>
                                using namespace std;

                                int main() {
                                    int n;
                                    cout << "Enter number of elements: ";
                                    cin >> n;
                                    int *arr = new int[n];
                                    cout << "Enter elements: ";
                                    for (int i = 0; i < n; i++) {
                                        cin >> arr[i];
                                    }
                                    cout << "Elements are: ";
                                    for (int i = 0; i < n; i++) {
                                        cout << arr[i] << " ";
                                    }
                                    cout << endl;
                                    delete[] arr;
                                    return 0;
                                }

    4. Function Pointers

  4. Write a C++ program to use function pointers to select different functions for addition, subtraction, multiplication, and division. 
                            Sample Output
                        
                            Addition: 5 + 3 = 8
                            Subtraction: 5 - 3 = 2
                            Multiplication: 5 * 3 = 15
                            Division: 6 / 3 = 2
                        
                    
                                
                                #include <iostream>
                                using namespace std;

                                int add(int a, int b) { return a + b; }
                                int subtract(int a, int b) { return a - b; }
                                int multiply(int a, int b) { return a * b; }
                                int divide(int a, int b) { return a / b; }

                                int main() {
                                    int (*funcPtr)(int, int);
                                    int a = 5, b = 3;
                                    
                                    funcPtr = add;
                                    cout << "Addition: " << a << " + " << b << " = " << funcPtr(a, b) << endl;
                                    
                                    funcPtr = subtract;
                                    cout << "Subtraction: " << a << " - " << b << " = " << funcPtr(a, b) << endl;
                                    
                                    funcPtr = multiply;
                                    cout << "Multiplication: " << a << " * " << b << " = " << funcPtr(a, b) << endl;
                                    
                                    int c = 6;
                                    funcPtr = divide;
                                    cout << "Division: " << c << " / " << b << " = " << funcPtr(c, b) << endl;
                                    
                                    return 0;
                                }

    5. Pointers to Structures

  5. Write a C++ program to use a pointer to a structure to modify structure members. 
                            Sample Output
                        
                            Before modification: Name = John, Age = 25
                            After modification: Name = Jane, Age = 30
                        
                    
                                
                                #include <iostream>
                                using namespace std;

                                struct Person {
                                    string name;
                                    int age;
                                };

                                void modifyPerson(Person *p) {
                                    p->name = "Jane";
                                    p->age = 30;
                                }

                                int main() {
                                    Person person = {"John", 25};
                                    cout << "Before modification: Name = " << person.name << ", Age = " << person.age << endl;
                                    modifyPerson(&person);
                                    cout << "After modification: Name = " << person.name << ", Age = " << person.age << endl;
                                    return 0;
                                }
